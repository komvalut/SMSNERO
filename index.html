async function initiatePurchase(amount, phone) { document.getElementById('payment-area').style.display = "block"; document.getElementById('loading-text').innerText = "Opening Swiss Bitcoin Pay..."; try { const res = await fetch('/api/make-invoice', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ amount: amount, memo: "SMS for " + phone, phone: phone }) }); const data = await res.json(); // AUTOMATSKI OTVARA SWISS PAY U NOVOM TABU window.open(data.payment_url, '_blank'); document.getElementById('loading-text').innerText = "Please complete the payment in the new tab. Waiting for confirmation..."; // Počinje da prati uplatu koristeći ID fakture monitorPayment(data.id, phone); } catch (e) { document.getElementById('loading-text').innerText = "Error connecting to Swiss Bitcoin Pay."; } }
